Terraform Private Subnet Creation Code


    resource "aws_subnet" "dev_subnet_private_1" {
      vpc_id                  = aws_vpc.dev_vpc.id
      cidr_block              = "10.20.5.0/24"

      map_public_ip_on_launch = false

        Name = "dev_subnet_private_1"
      }
    }
